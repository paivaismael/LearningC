#include<stdio.h>

int count(int n)
{
	int c;
	c=0;
	while (n!=0)
	{
	n=n/10;
	c=c+1;
	}
	return c;
}

int digit(int n,int m)
{
	int c;
	c=1;
	for (c<=m)
	{
	m=n%10;
	n=n/10;
	c=c+1;
	}
	return m;
}

int main()
{
	int n,m,i,p,z;
	printf("Please enter with a sequence of zeros and ones: ");
	scanf("%d",&n);
	z=0;
	m=count(n);
	i=1;
	for (i<=m)
	{
	p=digit(n,i);
	z=z+p*(2^(i-1));
	}
	printf("The sequence of zeros and ones %d represents the integer
%d.\n",n,z)
	return 0;
}
